102 	 * Checks to see whether the given identifier or Class is a xtend class. (Checks for .isClass)<br>
103 	 * NOTE: Classes that have not yet loaded all of their dependencies, will return FALSE for this check.
104 	 *
105 	 * @public
106 	 * @param		{String|Object}	identifier			The namespace identifier, the Class or an instance of the class.
107 	 * @returns		{Boolean}									Whether or not this is a Class.
108 	 */

110 	_xtend.isClass = function(identifier){
111 		
112 		if(typeof identifier === "string"){
113 			var parts = identifier.split(_separator);
114 			var identifierClass = window;
115 			
116 			for (var j = 0; j < parts.length; j++){
117 				if(!identifierClass[parts[j]]){
118 					return false;
119 				}
120 				identifierClass = identifierClass[parts[j]];
121 			}
122 		}
123 
124 		else{
125 			identifierClass = identifier;
126 		}
127 		
128 		if(typeof identifierClass === "object" || typeof identifierClass === "function"){
129 			return identifierClass.isClass;
130 		}
131 
132 		return false;
133 	}

135 	/**
136 	 * Gets the namespace/Class, or creates it if it does not exist. Also optionally creates Objects in the specified namepsace.
137 	 *
138 	 * @public
139 	 * @param		{String}				namespace			The namespace/package as a {String}
140 	 * @param		{Object}				[classes]			An object of class definitions which will be added to the namespace
141 	 * @returns		{Object}										The object that represents the namespace passed in as the first argument.
142 	 */
143 
144 	_xtend.namespace = function(namespace, classes){
145 		classes = classes || false;
146 		var ns = window;
147 
148 		if(namespace != '') {
149 			var parts = namespace.split(_separator);
150 			for (var i = 0; i < parts.length; i++) {
151 				if (!ns[parts[i]]){
152 					ns[parts[i]] = {};
153 				}
154 				ns = ns[parts[i]];
155 			}
156 		}
157 
158 		if(classes !== false){
159 			
160 			for(var className in classes){
161 				if(classes[className].superClassIdentifier){
162 					if(!classes.requires){classes.requires = [];}
163 					classes.requires.push(classes[className].superClassIdentifier);
164 					if(_extendQueue.indexOf(namespace + _separator + className) === -1){
165 						_extendQueue.push(namespace + _separator + className);
166 					}
167 				}
168 			}
169 
170 			if("requires" in classes){
171 				
172 				for(var className in classes){
173 					if(className !== "requires"){
174 						if(!_xtend.isClass(classes[className])){
175 							classes[className].dependencies = classes.requires;
176 							classes[className].nsID = namespace;
177 							classes[className].ns = ns;
178 							classes[className].className = className;
179 						}
180 						else{
181 							classes[className].prototype.dependencies = classes.requires;
182 							classes[className].prototype.nsID = namespace;
183 							classes[className].prototype.ns = ns;
184 							classes[className].prototype.className = className;
185 						}
186 
187 						ns[className] = classes[className];
188 					}
189 				}
190 				_xtend.require(classes.requires, function(){});
191 			}
192 		}
193 
194 		return ns;
195 	}

197 	/**
198 	 * Extends a given class asynchronously.
199 	 *
200 	 * @public
201 	 * @param		 {String}			identifier			The fully qualified class name of the class you want to extend.
202 	 * @param		 {Object}			obj					A Class Object
203 	 * @returns		 {Object}									The extended Class, or, if the Class to be extended has yet to be loaded in, the original Object with a few more properties.
204 	 */ 
205 
206 	_xtend.extend = function(identifier, obj){
207 	 
208 		// Let's check to see if the Class is already defined...
209 		var parts = identifier.split(_separator);
210 		var identifierClass = window;
211 		
212 		for (var j = 0; j < parts.length; j++){
213 			
214 			if(!identifierClass[parts[j]]){
215 				break;
216 			}
217 
218 			identifierClass = identifierClass[parts[j]];
219 		}
220 		
221 		// If the Class exists and is a xTend class, then return the extended object.
222 		if(identifierClass.isClass){
223 			obj = identifierClass.extend(obj);			
224 		}
225 
226 		else{
227 			obj.superClassIdentifier = identifier;
228 		}
229 
230 		obj.extendedFrom = identifier;
231 
232 		return obj;
233 	}

235 	/**
236 	 * Imports properties from the specified namespace to the global space (ie. under window)
237 	 *
238 	 * The identifier {String} can contain the * wildcard character as its last segment (eg: com.test.*) 
239 	 * which will import all properties from the namespace.
240 	 * 
241 	 * If not, the targeted namespace will be imported (ie. if com.test is imported, the test object 
242 	 * will now be global).
243 	 * 
244 	 * Only use xTend.use() if you know the Class/package has already been defined (loaded).
245 	 * Because xTend.use() imports objects/namespaces into the global namespace (window),
246 	 * it is highly recommended that you call xTend.unuse() to clean up afterwards.
247 	 *
248 	 * xTend.unuse() is automatically called after every Class method.
249 	 *
250 	 * @public
251 	 * @param	 {String|Array}		identifier		The namespace as a String or an Array of namespaces
252 	 */
253 	 
254 	_xtend.use = function(identifier){
255 		
256 		xTend.unuse(identifier);
257 
258 		var identifiers;
259 
260 		if(typeof(identifier) !== 'object' && !identifier.sort){
261 			identifiers = new Array(identifier);
262 		}
263 		else{
264 			identifiers = identifier;
265 		}
266 
267 		for (var i = 0; i < identifiers.length; i++) {
268 			identifier = identifiers[i];	
269 			
270 			var parts = identifier.split(_separator);
271 			var target = parts.pop();
272 			var ns = _xtend.namespace(parts.join(_separator));
273 			
274 			if (target === '*') {
275 				// imports all objects from the identifier, can't use include() in that case
276 				for (var objectName in ns) {
277 					_origWindowNS[objectName] = (window[objectName]) ? window[objectName] : null;
278 					window[objectName] = ns[objectName];
279 				}
280 			}
281 			else{
282 				// imports only one object
283 				if(ns[target]){
284 					_origWindowNS[target] = (window[target]) ? window[target] : null;
285 					window[target] = ns[target];
286 				}
287 			}
288 		}
289 	}

293 	 * The opposite of xTend.use() (who would'a thunk it!!)
294 	 *
295 	 * This method is automatically called after every Class method.
296 	 *
297 	 * @public
298 	 * @param	 {String|Array}		identifier		The namespace as a String or an Array of namespaces
299 	 */
300 
301 	_xtend.unuse = function(identifier){
302 
303 		identifier = identifier || "*";
304 		var identifiers;
305 
306 		if(typeof(identifier) !== 'object' && !identifier.sort){
307 			identifiers = new Array(identifier);
308 		}
309 		else{
310 			identifiers = identifier;
311 		}
312 
313 		for (var i = 0; i < identifiers.length; i++) {
314 			
315 			identifier = identifiers[i];
316 
317 			var parts = identifier.split(_separator);
318 			var target = parts.pop();
319 
320 			if(target === '*'){
321 				for(var prop in _origWindowNS){
322 					window[prop] = _origWindowNS[prop];
323 					if(window[prop] === null){
324 						delete window[prop];
325 					}
326 				}
327 			}
328 			else{
329 				// imports only one object
330 				if(_origWindowNS[target]){
331 					window[target] = _origWindowNS[target];
332 					if(window[target] === null){
333 						delete window[target];
334 					}
335 				}
336 			}
337 		}
338 	}

341 	/**
342 	 * Tells xTend that x file provides these Class definitions.
343 	 * Where x file = filePath and classes is an aray of the Classes that xFile defines.
344 	 * Useful in cases where you group specific things into minfiied js files.
345 	 *
346 	 * xTend.provides can load the file right away, by passing doLoad as true, and a callback function.
347 	 * Otherwise, it just maps the classes to the filePath for any subsequent calls to xTend.require() which might refer
348 	 * to said classes.
349 	 *
350 	 * @public
351 	 * @param	 {String}					filePath			The path to the JS file.
352 	 * @param	 {String|Array}			identifier		The namespace as a {String} or an {Array} of namespaces
353 	 */
354 
355 	_xtend.provides = function(filePath, classes, doLoad, callback){
356 
357 		// If classes is a String, create an array
358 		if(typeof(classes) !== 'object' && !classes.sort){
359 			classes = new Array(classes);
360 		}
361 
362 		for(var i = 0; i < classes.length; i ++){
363 			_classMappings[classes[i]] = filePath;
364 		}
365 
366 		if(doLoad){
367 			_xtend.require(classes, callback);
368 		}
369 	}

372 	/**
373 	 * Imports the packages/classes for a given identifier. Loads them if neccessary and includes them in the global namespace.
374 	 *
375 	 * @public
376 	 * @param	 {String|Array}		classes					The class(es) to load/include.
377 	 * @param	 {Function}				completeFunc			The function to call once all classes (and their dependencies) have been included/loaded.
378 	 */
379 
380 	_xtend.require = function(classes, completeFunc){
381 
382 		if(typeof classes === "string"){classes = [classes];}
383 		
384 		var classFiles = [];
385 		
386 		for(var i = 0; i < classes.length; i ++){
387 			var identifier = classes[i];
388 
389 			if((_loadedClasses.indexOf(identifier) === -1) && !_xtend.isClass(identifier)){	
390 				_loadedClasses.push(identifier);
391 				classFiles.push(_xtend.getIdentifierURL(identifier));
392 			}
393 		}
394 
395 		if(classFiles.length > 0){	
396 			var queue = {
397 				classes: classes,
398 				classFiles: classFiles,
399 				complete: completeFunc
400 			};
401 			
402 			_loadQueues.push(queue);
403 
404 			yepnope({
405 
406 				load : classFiles,
407 				callback : function(){
408 
409 					for(var i = _extendQueue.length - 1; i >= 0; i --){
410 						
411 						var packageArray = _extendQueue[i].split(_separator);
412 						var className	= packageArray.splice(packageArray.length-1, 1);
413 						var namespace = packageArray.join(_separator);
414 						
415 						var packageObj = _xtend.namespace(namespace);
416 						
417 						if(typeof packageObj[className].superClassIdentifier !== "undefined"){
418 							var superClass = _xtend.namespace(packageObj[className].superClassIdentifier);
419 							if(superClass.isClass){
420 								
421 								var dependencies = packageObj[className].dependencies;
422 								var scID = packageObj[className].superClassIdentifier;
423 
424 								packageObj[className].superClassIdentifier = null;
425 								delete packageObj[className].superClassIdentifier;
426 								
427 								packageObj[className] = superClass.extend(packageObj[className]);
428 								
429 								packageObj[className].dependencies = dependencies;
430 
431 								_extendQueue.splice(i, 1);
432 
433 							}
434 						}
435 					}
436 
437 					for(var i = _loadQueues.length -1; i >= 0; i --){
438 439 						var queue = _loadQueues[i];
440 						var dependenciesLoaded = true;
441 						
442 						for(var j = 0; j < _loadQueues[i].classes.length; j ++){
443 							
444 							if(!_xtend.isClass(_loadQueues[i].classes[j])){
445 								dependenciesLoaded = false;
446 								break;
447 							}
448 
449 							var obj = _xtend.namespace(_loadQueues[i].classes[j]);
450 							
451 							if(obj.dependencies){
452 								for(var k = 0; k < obj.dependencies.length; k ++){
453 									if(_loadedClasses.indexOf(obj.dependencies[i]) === -1){
454 										dependenciesLoaded = false;
455 										break;
456 									}
457 								}
458 							}
459 							if(!dependenciesLoaded){break;}
460 						}
461 						if(dependenciesLoaded){
462 							queue.complete.call();
463 							_loadQueues.splice(i, 1);
464 						}
465 					}
466 				}
467 			});
468 469 		}
470 
471 		else{
472 
473 			completeFunc.call();
474 		}
475 	}
